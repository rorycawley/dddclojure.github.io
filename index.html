recur - recursion


  (defn print-down-from [x]
    (when (pos? x) ; when positive
      (println x) ; side effect
      (recur (dec x)))) ; loop x - 1
   (print-down-from 3)

recur - recursion


  (defn sum-down-from [sum x]
    (if (pos? x)
      (recur (+ sum x) (dec x))
        sum))
  (sum-down-from 0 10)

loop


  (defn sum-down-from [initial-x]
    (loop [sum 0, x initial-x]
      (if (pos? x)
        (recur (+ sum x) (dec x))
       sum)))
  (sum-down-from 0 10)
