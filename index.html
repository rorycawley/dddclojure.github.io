Use defprotocol

  
(ns our.collection)

(defprotocol OurCollection
  (count [this])
  (get [this k] [this k not-found]))

(defn empty?
  [this]
  (zero? (count this)))

Require a namespace and rename some variables 


(ns my.ns
  (:require
      [clojure.set :refer [union intersection] :rename {union UNION}]))
      
  [(intersection #{1 2 3} #{2 3 4})
   (UNION #{1 2 3} #{2 3 4 5})]
